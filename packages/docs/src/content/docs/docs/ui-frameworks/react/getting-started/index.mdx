<Steps>
    <ol>
        <li>
            ### Install
            <Tabs>
                <TabItem label="npm" icon="npm">
                    ```bash
                    npm install @navnex-kit/core @navnex-kit/react
                    ```
                </TabItem>
                <TabItem label="pnpm" icon="pnpm">
                    ```bash
                    pnpm add @navnex-kit/core @navnex-kit/react
                    ```
                </TabItem>
            </Tabs>
        </li>

        <li>
            ### Setup provider
            Wrap your application with the `NavnexProvider` to enable all navnex-kit components.
            This should be done at the layout level to ensure global access.

            > Recommended: Place this in `src/main.tsx` (React root element)

            <FileTree>
                - src
                    - main.tsx
            </FileTree>

            ```tsx
            import { StrictMode } from "react"
            import { createRoot } from "react-dom/client"
            import App from "./App"
            import { NavnexProvider } from "@navnex-kit/react"

            createRoot(document.getElementById("root")!).render(
                <StrictMode>
                    <NavnexProvider>
                        <App />
                    </NavnexProvider>
                </StrictMode>
            )
            ```
        </li>

        <li>
            ### Enjoy!
            <Tabs>
                <TabItem label="Usage">
                    ```tsx
                    import { Button } from "@navnex-kit/react"

                    function App() {
                        return (
                            <>
                                <Button>
                                    Click me
                                </Button>
                            </>
                        )
                    }

                    export default App
                    ```
                </TabItem>
                <TabItem label="Preview">
                    <Button>
                        Click me
                    </Button>
                </TabItem>
            </Tabs>
        </li>
    </ol>

</Steps>
