import { Button } from "@navnex-kit/svelte"
import { Tabs, TabItem } from "@astrojs/starlight/components"
import { Steps } from "@astrojs/starlight/components"
import { FileTree } from "@astrojs/starlight/components"

<Steps>
    <ol>
        <li>
            ### Install
            <Tabs>
                <TabItem label="npm" icon="npm">
                    ```bash
                    npm install @navnex-kit/core @navnex-kit/svelte
                    ```
                </TabItem>
                <TabItem label="pnpm" icon="pnpm">
                    ```bash
                    pnpm add @navnex-kit/core @navnex-kit/svelte
                    ```
                </TabItem>
            </Tabs>
        </li>

        <li>
            ### Setup provider
            Wrap your application with the `NavnexProvider` to enable all navnex-kit components.
            This should be done at the layout level to ensure global access.

            > Recommended: Place this in `src/routes/+layout.svelte` (SvelteKit root layout)

            <FileTree>
                - src
                    - routes
                        - +layout.svelte
            </FileTree>

            ```svelte
            <script lang="ts">
                import { NavnexProvider } from "@navnex-kit/svelte"
            </script>

            <NavnexProvider>
                {@render children()}
            </NavnexProvider>
            ```
        </li>

        <li>
            ### Enjoy!
            <Tabs>
                <TabItem label="Usage">
                    ```svelte
                    <script lang="ts">
                        import { Button } from "@navnex-kit/svelte"
                    </script>

                    <Button>
                        Click me
                    </Button>
                    ```
                </TabItem>
                <TabItem label="Preview">
                    <Button>
                        Click me
                    </Button>
                </TabItem>
            </Tabs>
        </li>
    </ol>

</Steps>
