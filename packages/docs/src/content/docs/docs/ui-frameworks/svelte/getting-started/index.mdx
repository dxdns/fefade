title: Installation
description: How to install and set up Feflow UI in your project
---

import { Button } from "@feflow-ui/svelte"
import { Tabs, TabItem } from "@astrojs/starlight/components"
import { Steps } from "@astrojs/starlight/components"
import { FileTree } from "@astrojs/starlight/components"

<Steps>
    <ol>
        <li>
            ### Install
            <Tabs>
                <TabItem label="npm" icon="npm">
                    ```bash
                    npm install @feflow-ui/core @feflow-ui/svelte
                    ```
                </TabItem>
                <TabItem label="pnpm" icon="pnpm">
                    ```bash
                    pnpm add @feflow-ui/core @feflow-ui/svelte
                    ```
                </TabItem>
            </Tabs>
        </li>

        <li>
            ### Setup provider
            Wrap your application with the `FeflowProvider` to enable all Feflow UI components.
            This should be done at the layout level to ensure global access.

            > Recommended: Place this in `src/routes/+layout.svelte` (SvelteKit root layout)

            <FileTree>
                - src
                    - routes
                        - +layout.svelte
            </FileTree>

            ```svelte
            <script lang="ts">
                import { FeflowProvider } from "@feflow-ui/svelte"
            </script>

            <FeflowProvider>
                {@render children()}
            </FeflowProvider>
            ```
        </li>

        <li>
            ### Enjoy!
            <Tabs>
                <TabItem label="Usage">
                    ```svelte
                    <script lang="ts">
                        import { Button } from "@feflow-ui/svelte"
                    </script>

                    <Button>
                        Click me
                    </Button>
                    ```
                </TabItem>
                <TabItem label="Preview">
                    <Button>
                        Click me
                    </Button>
                </TabItem>
            </Tabs>
        </li>
    </ol>

</Steps>
