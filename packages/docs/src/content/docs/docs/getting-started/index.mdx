import { Button } from "@dxdns/feflow-svelte"
import { Tabs, TabItem } from "@astrojs/starlight/components"
import { Steps } from "@astrojs/starlight/components"
import { FileTree } from "@astrojs/starlight/components"

<Steps>
    <ol>
        <li>
            ### Install
            <Tabs>
                <TabItem label="Svelte" icon="seti:svelte">
                    ```bash
                    npm install @dxdns/feflow-svelte
                    ```
                </TabItem>
                <TabItem label="React" icon="seti:react">
                    ```bash
                    npm install @dxdns/feflow-react
                    ```
                </TabItem>
            </Tabs>
        </li>

        <li>
            ### Setup provider
            Wrap your application with the `FeflowProvider` to enable all Feflow components.
            This should be done at the layout level to ensure global access.

            <Tabs>
                <TabItem label="Svelte" icon="seti:svelte">
                    > Recommended: Place this in `src/routes/+layout.svelte` (SvelteKit root layout)

                    <FileTree>
                        - src
                            - routes
                                - +layout.svelte
                    </FileTree>

                    ```svelte
                    <script lang="ts">
                        import { FeflowProvider } from "@dxdns/feflow-svelte"
                    </script>

                    <FeflowProvider>
                        {@render children()}
                    </FeflowProvider>
                    ```
                </TabItem>
                <TabItem label="React" icon="seti:react">
                    > Recommended: Place this in `src/main.tsx` (React root element)

                    <FileTree>
                        - src
                            - main.tsx
                    </FileTree>

                    ```tsx
                    import { StrictMode } from "react"
                    import { createRoot } from "react-dom/client"
                    import App from "./App"
                    import { FeflowProvider } from "@dxdns/feflow-react"

                    createRoot(document.getElementById("root")!).render(
                        <StrictMode>
                            <FeflowProvider>
                                <App />
                            </FeflowProvider>
                        </StrictMode>
                    )
                    ```
                </TabItem>
            </Tabs>
        </li>

        <li>
            ### Enjoy!
            <Tabs>
                <TabItem label="Svelte" icon="seti:svelte">
                    ```svelte
                    <script lang="ts">
                        import { Button } from "@dxdns/feflow-svelte"
                    </script>

                    <Button>
                        Click me
                    </Button>
                    ```
                </TabItem>
                <TabItem label="React" icon="seti:react">
                    ```tsx
                    import { Button } from "@dxdns/feflow-react"

                    function App() {
                        return (
                            <>
                                <Button>
                                    Click me
                                </Button>
                            </>
                        )
                    }

                    export default App
                    ```
                </TabItem>
                <TabItem label="Preview">
                    <Button>
                        Click me
                    </Button>
                </TabItem>
            </Tabs>
        </li>
    </ol>

</Steps>
